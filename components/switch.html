# 基础用法

<template>
  <div>
    <v-switch v-model="value" @change="handleChange"></v-switch>
  </div>
</template>

<script>
export default {
  data() {
    return {
      value: false
    }
  },
  methods: {
    handleChange (value) {
      console.log(value)
    }
  }
}
</script>

# 设置背景色
